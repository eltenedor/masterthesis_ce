\begin{frame}

\frametitle{Motivation}
    \vspace{0.5cm}
    Herausforderungen für CFD-Applikationen:
    \begin{itemize}
    \item \small{Schnelle Verfügbarkeit von Simulationsergebnissen}
    \item \small{Ergebnisse mit hoher Genauigkeit}
    \end{itemize}

    \begin{columns}
        \begin{column}{6cm}
            \begin{figure}
            \centering
            \includegraphics[width= 0.8\linewidth]{./img/turbine.jpg}
            \caption{Gasturbine (VDI)}
            \end{figure}
        \end{column}
        \begin{column}{6cm}
            \begin{itemize}
                \item \small{Einsatz robuster Algorithmen}
                \item \small{Skalierbarkeit der Lösungsmethode}
                \item \small{Effizienzsteigerung durch Adaptivität}
            \end{itemize}
        \end{column}
    \end{columns}

\end{frame}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}

\frametitle{Motivation}
    Vollständig gekoppelter Lösungsansatz für Navier-Stokes Gleichungen (Darwish 2009):
    \begin{itemize}
    \item Semi-implizite Druck-Geschwindigkeits Kopplung statt sequentieller Lösung
    \item Robuster Algorithmus ohne Unterrelaxation
    \end{itemize}

    \vspace{1cm}
    Neue Herausforderung:
    \begin{itemize}
      \item Umgang mit Speicheranforderungen
      \item Auswahl geeigneter skalierbarer Gleichungslöser
    \end{itemize}

\end{frame}

\section{Aufgabenstellung und Bearbeitung}
\begin{frame}

\frametitle{Aufgabenstellung und Bearbeitung}
  \begin{itemize}
    \item Implementierung eines vollständig gekoppelten Lösungsansatzes 
      \begin{itemize}\scriptsize
        \item Finite-Volumen Diskretisierung der 3d Navier-Stokes Gleichungen auf block-strukturierten, lokal verfeinerten Gittern mit hängenden Knoten
        \item Kopplungsansätze für Temperaturgleichung 
        \item MPI-Parallelisierung des Lösungsansatzes mit PETSc
      \end{itemize}
    \item Skalierbarkeitsuntersuchung auf HHLR
    \item Performancevergleich mit herkömmlichem SIMPLE Verfahren für unterschiedliche Testfälle
      \begin{itemize}\scriptsize
        \item Manufactured Solution
        \item Kanalströmung mit komplexem Hindernis
        \item Strömung in beheizter Kavität (Adaption MIT - Benchmark)
      \end{itemize}
  \end{itemize}

\end{frame}
